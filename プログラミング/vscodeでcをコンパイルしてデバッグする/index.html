    # コンパイラはgccを使う
CC = gcc
# コンパイルオプション: -Wallは全ての警告を表示する, -std=c89はC89標準に準拠する,
# -pedantic-errorsは警告をエラーとして扱う, -Werrorは警告をエラーとして扱う,
# -gはデバッグ情報を付加する
CFLAGS = -Wall -std=c89 -pedantic-errors -Werror -g
TARGET = a.out
SRCS = $(shell find . -name "*.c")
OBJS = $(SRCS:.c=.o)

all: $(TARGET)

$(TARGET): $(OBJS)
    $(CC) $(CFLAGS) -o $(TARGET) $(OBJS)

%.o: %.c
    $(CC) $(CFLAGS) -c $< -o $@

clean:
    rm -f $(OBJS) $(TARGET)
